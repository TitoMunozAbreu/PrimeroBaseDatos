DROP database IF EXISTS oficina;
CREATE DATABASE oficina;

USE oficina;


DROP TABLE IF EXISTS EMPLEADOS;
DROP TABLE IF EXISTS DEPARTAMENTOS;

CREATE TABLE DEPARTAMENTOS (
DEPT_NO INT PRIMARY KEY,
NOMBRE VARCHAR(15),
LOCALIDAD VARCHAR(15)
) ENGINE = InnoDB;

CREATE TABLE EMPLEADOS (
EMP_NO INT PRIMARY KEY,
NOMBRE VARCHAR(15),
OFICIO VARCHAR(15),
DEPT_NO INT NOT NULL,
SALARIO INT,
ALTA DATE,
INDEX deptno_ind (dept_no),
FOREIGN KEY (dept_no) REFERENCES DEPARTAMENTOS (dept_no)
) ENGINE = InnoDB;

DROP TABLE IF EXISTS EMPLEADOS_ANTIGUOS;
CREATE TABLE EMPLEADOS_ANTIGUOS (
EMP_NO INT PRIMARY KEY,
NOMBRE VARCHAR(15),
SALARIO INT );

DROP TABLE IF EXISTS EMPLEADOS_NUEVOS;
CREATE TABLE EMPLEADOS_NUEVOS (
EMP_NO INT PRIMARY KEY,
NOMBRE VARCHAR(15),
SALARIO INT );
